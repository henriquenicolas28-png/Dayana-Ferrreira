  <script>
    function buscarTalentos() {
      const curso = document.getElementById('curso').value.toLowerCase();
      const habilidades = document.getElementById('habilidades').value.toLowerCase();
      const disponibilidade = document.getElementById('disponibilidade').value;

      // Simulação de dados
      const estudantes = [
        {
          nome: "Dayana Silva",
          curso: "Gestão Empresarial",
          habilidades: "Excel, planejamento, comunicação",
          disponibilidade: "estagio"
        },
        {
          nome: "Carlos Mendes",
          curso: "Análise e Desenvolvimento de Sistemas",
          habilidades: "JavaScript, HTML, CSS",
          disponibilidade: "tempo-integral"
        }
      ];

      const resultados = estudantes.filter(e => {
        return (
          (!curso || e.curso.toLowerCase().includes(curso)) &&
          (!habilidades || e.habilidades.toLowerCase().includes(habilidades)) &&
          (!disponibilidade || e.disponibilidade === disponibilidade)
        );
      });

      const container = document.getElementById('resultados');
      container.innerHTML = '';

      if (resultados.length === 0) {
        container.innerHTML = '<p>Nenhum estudante encontrado com os critérios selecionados.</p>';
      } else {
        resultados.forEach(e => {
          container.innerHTML += `
            <div class="result">
              <strong>${e.nome}</strong><br/>
              Curso: ${e.curso}<br/>
              Habilidades: ${e.habilidades}<br/>
              Disponibilidade: ${e.disponibilidade}<br/>
              <button>Ver Perfil</button>
            </div>
          `;
        });
      }
    }
  </script>
